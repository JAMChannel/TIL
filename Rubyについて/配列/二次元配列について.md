# 2次元配列を作る
```
team_c = ["勇者", "戦士", "魔法使い"]
team_d = ["盗賊", "忍者", "商人"]
team_e = ["スライム", "ドラゴン", "魔王"]
teams = [team_c, team_d, team_e]
p teams

p teams[0]
p teams[0][0]
p teams[0][1]
p teams[0][2]

```


# 2次元配列の基本操作1
 ```
teams = [["勇者", "戦士", "魔法使い"], ["盗賊", "忍者", "商人"], ["スライム", "ドラゴン", "魔王"]]
p teams


p teams[0]         # ["勇者", "戦士", "魔法使い"]
p teams[0][0]      # "勇者"
p teams[0][1]      # "戦士"
p teams[0][2] 

teams[0][0] = "魔導士"
p teams
```


# 2次元配列の基本操作 追加、削除
```
teams = [["勇者", "戦士"], ["盗賊", "忍者", "商人"], ["スライム", "ドラゴン", "魔王"], ["魔法使い"]]
p teams


#  追加
teams.push(["メタルモンスター","シルバーモンスター","ブラックモンスター"])
p teams


teams[0].push("レッドドラゴン")
p teams

# 削除する方法
teams.delete_at(1)
p teams
teams[0].delete_at(1)
p teams
```
